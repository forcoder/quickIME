
### 使用示例
```cpp
#include "quickinput/ai/inference_engine.h"

qi::ai::InferenceEngine engine;
if (engine.Initialize(L"model.gguf")) {
    auto suggestions = engine.GenerateSuggestions(
        L"上下文", L"用户输入", 3);
    // 处理suggestions...
}
```
